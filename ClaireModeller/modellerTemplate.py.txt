# Homology modeling by the automodel class
from modeller import *              # Load standard Modeller classes
from modeller.automodel import *    # Load the automodel class

# Redefine the special_patches routine to include the additional disulfides
# (this routine is empty by default):
class MyModel(automodel):
    def special_patches(self, aln):
        # A disulfide between residues 121 and 216:
        self.patch(residue_type='DISU', residues=(self.residues['97'],
                                                  self.residues['179']))

log.verbose()    # request verbose output
env = environ()  # create a new MODELLER environment to build this model in

# directories for input atom files
env.io.atom_files_directory = ['/Users/jeevantewari/Desktop/ClaireModeller']

a = MyModel(env,
            alnfile  = 'ali.txt',     # alignment filename
            knowns   = ('1U19A','3ODUA','2YDV','2LNL'),              # codes of the templates
            sequence = 'OR1A1')              # code of the target
a.starting_model= 1                 # index of the first model
a.ending_model  = 5                 # index of the last model
                                    # (determines how many models to calculate)
a.make()                            # do the actual homology modeling
